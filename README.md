<script>
  // Function to switch language
  function switchLang(lang) {
    // 1. Hide all elements with class 'lang'
    var langElements = document.querySelectorAll('.lang');
    langElements.forEach(function(el) {
      el.style.display = 'none';
    });

    // 2. Show only the elements for the selected language
    var selectedLangElements = document.querySelectorAll('.lang.' + lang);
    selectedLangElements.forEach(function(el) {
      el.style.display = 'block';
    });

    // 3. Save the user's preference in localStorage
    localStorage.setItem('preferredLang', lang);
  }

  // Function to set the initial language on page load
  function setInitialLang() {
    var preferredLang = localStorage.getItem('preferredLang');
    
    // If no preference is saved, default to English
    if (!preferredLang) {
      preferredLang = 'en';
    }
    
    switchLang(preferredLang);
  }

  // Call the function when the page loads
  document.addEventListener('DOMContentLoaded', setInitialLang);
</script>
